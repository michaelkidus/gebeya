                    imagePromises.push(promise);
                }
            }
            
            Promise.all(imagePromises).then(imageData => {
                ad.images = imageData;
                saveAd(ad);
            });
        } else {
            saveAd(ad);
        }
    });

    // Create ad card element
    function createAdCard(ad) {
        const card = document.createElement('div');
        card.className = 'listing-card bg-white rounded-lg shadow-md overflow-hidden transition duration-300';
        
        let statusClass = 'status-active';
        if (ad.status === 'pending') {
            statusClass = 'status-pending';
        } else if (ad.status === 'sold') {
            statusClass = 'status-sold';
        }
        
        const statusText = ad.status === 'active' ? 'ንቁ' : 
                          ad.status === 'pending' ? 'በመጠባበቅ ላይ' : 'የተሸጠ';
        
        card.innerHTML = `
            <div class="relative">
                ${ad.images.length > 0 ? 
                    `<img src="${ad.images[0]}" alt="${ad.title}" class="w-full h-48 object-cover">` : 
                    `<div class="w-full h-48 bg-gray-100 flex items-center justify-center">
                        <i class="fas fa-image text-4xl text-gray-400"></i>
                    </div>`}
                <span class="ad-status ${statusClass}">${statusText}</span>
            </div>
            <div class="p-4">
                <h3 class="font-bold text-lg text-gray-800 mb-1 amharic-font">${ad.title}</h3>
                <p class="text-gray-600 text-sm mb-2 amharic-font">${ad.location}</p>
                <div class="flex justify-between items-center">
                    <span class="font-bold text-orange-600 amharic-font">${ad.price} ብር</span>
                    <span class="text-gray-500 text-sm amharic-font"><i class="far fa-eye mr-1"></i> ${ad.views}</span>
                </div>
                ${myAdsSection.contains(card) ? `
                <div class="ad-actions">
                    <button class="ad-action-btn edit-btn amharic-font" data-id="${ad.id}">አርትዕ</button>
                    <button class="ad-action-btn delete-btn amharic-font" data-id="${ad.id}">ሰርዝ</button>
                </div>
                ` : ''}
            </div>
        `;
        
        return card;
    }
